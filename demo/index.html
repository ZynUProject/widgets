ZynU Widgets

Embeddable video player and subscribe button for the ZynU platform.

Video Player

<div id="zynu-player" style="width:640px;height:360px"></div>

<script type="module">
  import { createPlayer } from "@zynu/widgets";

  createPlayer({
    containerId: "zynu-player",
    videoId: "demo-video-id",
    autoplay: false,
    theme: "dark",
    onReady: () => console.log("Player ready"),
    onPlay:  () => console.log("Playing"),
  });
</script>

Subscribe Button

<div id="zynu-subscribe"></div>

<script type="module">
  import { createSubscribeButton } from "@zynu/widgets";

  createSubscribeButton({
    containerId: "zynu-subscribe",
    channelId: "channel-id",
    apiKey: "zynu_your_api_key",
    size: "md",
  });
</script>
